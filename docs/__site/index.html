Multi curves

x  = range(-1, 1, length=250)
y1 = @. exp(-x^2) * sinc(5x)
y2 = @. exp(-x^3) * sinc(3x)
y3 = @. exp(-x^4) * sinc(2x)
plot(x, y1, y2, y3)
gcf()

Adding curves

x  = range(-3, 3, length=250)
y  = @. exp(-x^2)
x2 = range(-3, 3, length=50)
y2  = @. -abs(x2/3) + 1
plot(x, y)
plot!(x2, y2)
gcf()
